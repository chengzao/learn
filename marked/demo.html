 Array.prototype.unique = function () {
    var json = {};
    var result = [];
    this.forEach(function (value) {
      // 判断数组中的类型
      var type = Object.prototype.toString.call(value).match(/\s(\w+)/)[1].toLowerCase();
      // 根据类型判断该数组中的项是否在json中
      if (!((type + '-' + value) in json)) {
        json[type + '-' + value] = true;
        result.push(value);
      }
    })
    return result;
  }
